class Solution:
    def countStudents(self, students: List[int], sandwiches: List[int]) -> int:
        t = [0] * 2
        for student in students:
            t[student] += 1
        while sandwiches and t[sandwiches[0]] > 0:
            while students[0] != sandwiches[0]:
                students.append(students.pop(0))
            students.pop(0)
            t[sandwiches[0]] -= 1
            sandwiches.pop(0)
        return len(students)

def averageWaitingTime(self, customers: List[List[int]]) -> float:
        n = len(customers)
        cook = 0
        wait = 0
        for arrival, duration in customers:
            if arrival >= cook:
                wait += duration
            else:
                wait += cook + duration - arrival
            cook = max(cook, arrival) + duration
        return wait / n

string maximumBinaryString(string s) {
        //zhao di yi ge 0
        int k = 0;
        while (k < s.size() && s[k] == '1') k ++;
        if (k == s.size()) return s;
        int cnt = 0;
        for (int i = k + 1; i < s.size(); i++ )
            if (s[i] == '0')
                cnt++;
        string res(s.size(), '1');
        res[k + cnt] = '0';
        return res;
    }

int minMoves(vector<int>& nums, int k) {
        vector<int> a;
        for (int i = 0; i < nums.size(); i ++)
            if (nums[i])
                a.push_back(i - a.size());
        //1 de wei zhi
        int n = a.size();
        typedef long long LL;
        vector<LL> s(n + 1);
        for (int i = 1; i <= n; i ++) s[i] = s[i - 1] + a[i - 1];
        LL res = 1e18;
        for (int i = k; i <= n; i++){
            int l = i - k + 1, r = i;
            int mid = (l + r) / 2;
            LL x = a[mid - 1];
            LL left = x * (mid - l) - (s[mid - 1] - s[l - 1]);
            LL right = (s[r] - s[mid]) - x * (r - mid);
            res = min(res, left + right);
        }
        return res;

    }

#3 石子游戏
（区间DP +博弈论:$\color{blue}{最坏情况下的最好选择}$）
定义F[i][j]为[i,j] 先手 - 后手 的最大分值
①取i： S[i + 1, j] - f[i + 1, j] // f[i + 1, j] 即(B - A )max
②取j： S[i, j - 1] - f[i, j - 1]

int stoneGameVII(vector<int>& stones) {
        const int n = stones.size();
        vector<int> sum(n + 1, 0); //定义前缀和数组
        for (int i = 1; i <= n; i ++) sum[i] = sum[i - 1] + stones[i - 1];
        vector<vector<int>> f(n + 1, vector<int>(n + 1)); //状态方程
        for (int len = 2; len <= n; len ++){
            for (int i = 1; i + len - 1 <= n; i++){
                int j = i + len - 1;
                f[i][j] = max(sum[j] - sum[i] - f[i + 1][j], sum[j - 1] - sum[i - 1] - f[i][j - 1]);
            }
        }
        return f[1][n];
    }

#4
$\color{cyan}{找最长上升子序列}$
设状态 $f(i)$ 表示以第 $i$ 个长方体为结尾的最大堆叠高度。
转移时，对于每个 $i$，枚举 $0≤j<i$。在满足要求后，$f(i)=max(f(i),f(j)+cuboids[i][2])$。

int maxHeight(vector<vector<int>>& cuboids) {
        const int n = cuboids.size();
        for (int i = 0; i < n; i++) sort(cuboids[i].begin(), cuboids[i].end());
        sort(cuboids.begin(), cuboids.end());
        vector<int> f(n);
        int ans = 0;
        for (int i = 0; i < n; i++){
            f[i] = cuboids[i][2]; //①初始化
            for (int j = 0;j < i; j++)
                if (cuboids[i][1] >= cuboids[j][1] && cuboids[i][2] >= cuboids[j][2])
                    f[i] = max(f[i], f[j] + cuboids[i][2]); //②状态转移
            ans = max(ans, f[i]);
        }
        return ans;
    }

def findBall(self, grid: List[List[int]]) -> List[int]:
        n, m = len(grid), len(grid[0])
        res = []
        for s in range(m):
            x = s
            for i in range(n):
                if grid[i][x] == 1:
                    if x == m - 1 or grid[i][x + 1] == -1:
                        x = -1
                        break
                    else:
                        x += 1
                else:
                    if x == 0 or grid[i][x - 1] == 1:
                        x = -1
                        break
                    else:
                        x -= 1
            res.append(x)
        return res

        s = {}
        ans = 0
        for i in deliciousness:
            for j in range(0, 22):
                try:
                    ans += s[2 ** j - i]

                s[i] = 1

            left = pre[i]

            idx1 = max(idx1, i + 1)
            r = pre[-1] - left
            mi = r // 2
            idx2 = bisect.bisect_right(pre, left + mi)
            idx2 = min(idx2, n)

    int waysToSplit(vector<int>& nums) {
        int n = nums.size(), mod = 1e9 + 7;
        vector<int> s(n+1);
        for(int i = 1; i <= n; i++) s[i] = s[i - 1] + nums[i - 1];
        int res = 0;
        for (int i = 3, j = 2, k = 2; i <= n; i++){
            while (s[n] - s[i -1] < s[i - 1] - s[j - 1]) j++;
            while (k + 1 < i && s[i - 1] - s[k] >= s[k]) k++; 
            if (j <= k && s[n] - s[i - 1] >= s[i - 1] - s[j - 1] && s[i - 1] - s[k - 1] >= s[k - 1])
                res = (res + k - j + 1) % mod;
        }
        return res;
    }
